  <script>
    // ── Poll Data ──────────────────────────────────────────────────
    const POLLS = [
      {
        id: "q1",
        question: "Who would you vote for if elections were tomorrow?",
        options: [
          { id: "vance",   label: "JD Vance"    },
          { id: "newsom",  label: "G. Newsom"   },
          { id: "unsure",  label: "Don't know"  },
        ]
      },
      {
        id: "q2",
        question: "Should voter ID be required by federal law?",
        options: [
          { id: "yes",     label: "Yes"          },
          { id: "no",      label: "No"           },
          { id: "neutral", label: "I don't care" },
        ]
      },
      {
        id: "q3",
        question: "Do you support US military intervention in Iran?",
        options: [
          { id: "yes",     label: "Yes"          },
          { id: "no",      label: "No"           },
          { id: "neutral", label: "I don't care" },
        ]
      },
    ];

    // ── State helpers ──────────────────────────────────────────────
    function loadState() {
      try { return JSON.parse(localStorage.getItem("wtp_polls") || "{}"); }
      catch { return {}; }
    }
    function saveState(state) {
      localStorage.setItem("wtp_polls", JSON.stringify(state));
    }

    let state = loadState();
    // state[pollId] = { voted: optionId | null, counts: { optId: n, ... } }

    function getOrInitPoll(pollId, options) {
      if (!state[pollId]) {
        state[pollId] = { voted: null, counts: {} };
        options.forEach(o => state[pollId].counts[o.id] = 0);
      }
      return state[pollId];
    }

    // ── Render ─────────────────────────────────────────────────────
    function render() {
      const container = document.getElementById("polls-container");
      container.innerHTML = "";

      POLLS.forEach((poll, idx) => {
        const ps = getOrInitPoll(poll.id, poll.options);
        const total = poll.options.reduce((s, o) => s + (ps.counts[o.id] || 0), 0);
        const hasVoted = ps.voted !== null;

        const card = document.createElement("div");
        card.className = "poll-card";
        card.innerHTML = `
          <p class="poll-number">Poll ${idx + 1} of ${POLLS.length}</p>
          <h2 class="poll-question">${poll.question}</h2>
          <div class="options" id="opts-${poll.id}"></div>
          <div class="results-section">
            <p class="results-label">Results</p>
            <div id="bars-${poll.id}"></div>
            <p class="total-votes">${total} vote${total !== 1 ? "s" : ""} cast</p>
            ${hasVoted ? `<p class="voted-notice">✓ You voted · change your vote any time</p>` : ""}
          </div>
        `;
        container.appendChild(card);

        // Buttons
        const optsEl = card.querySelector(`#opts-${poll.id}`);
        poll.options.forEach(opt => {
          const btn = document.createElement("button");
          btn.className = "vote-btn" + (ps.voted === opt.id ? " selected" : "");
          btn.textContent = opt.label;
          btn.addEventListener("click", () => castVote(poll.id, opt.id, poll.options));
          optsEl.appendChild(btn);
        });

        // Bars
        renderBars(poll, ps, total);
      });
    }

    function renderBars(poll, ps, total) {
      const barsEl = document.getElementById(`bars-${poll.id}`);
      if (!barsEl) return;
      barsEl.innerHTML = "";
      poll.options.forEach(opt => {
        const votes = ps.counts[opt.id] || 0;
        const pct = total > 0 ? Math.round((votes / total) * 100) : 0;
        const isSelected = ps.voted === opt.id;
        const row = document.createElement("div");
        row.className = "result-row";
        row.innerHTML = `
          <div class="result-meta">
            <span>${opt.label}</span>
            <span class="pct">${pct}% <span style="opacity:.5;font-weight:300">(${votes})</span></span>
          </div>
          <div class="bar-track">
            <div class="bar-fill ${isSelected ? "selected-bar" : ""}" style="width:${pct}%"></div>
          </div>
        `;
        barsEl.appendChild(row);
      });
    }

    function castVote(pollId, optId, options) {
      const ps = getOrInitPoll(pollId, options);

      // Undo previous vote if changing
      if (ps.voted !== null && ps.voted !== optId) {
        ps.counts[ps.voted] = Math.max(0, (ps.counts[ps.voted] || 0) - 1);
      }
      if (ps.voted !== optId) {
        ps.counts[optId] = (ps.counts[optId] || 0) + 1;
        ps.voted = optId;
      }

      saveState(state);
      render();
    }

    // ── Reset ──────────────────────────────────────────────────────
    document.getElementById("reset-button").addEventListener("click", () => {
      state = {};
      localStorage.removeItem("wtp_polls");
      render();
    });

    // ── Init ───────────────────────────────────────────────────────
    render();
  </script>
